import streamlit as st
import pandas as pd
from prophet import Prophet
import plotly.express as px
import speech_recognition as sr
import pyttsx3

# FIXED: Your exact filename from GitHub
@st.cache_data
def load_data():
    data = pd.read_excel("hyperlocal_demand_forecasting_with_grocery_items-2.xlsx")
    data['Month'] = pd.to_datetime(data['Month'])
    data['Product Name'] = data['Product Name'].astype(str)
    return data.sort_values('Month')

data = load_data()

# Seasons/festivals
seasons = {
    "None": ["January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"],
    "Diwali": ["October", "November"],
    "Christmas": ["December"],
    "Summer": ["April", "May", "June"],
    "Monsoon": ["July", "August", "September"],
    "New Year": ["January"]
}

# Sidebar
st.sidebar.title("Product Demand Forecasting")
selected_product = st.sidebar.selectbox("Select product", data['Product Name'].unique())
selected_season = st.sidebar.selectbox("Select season/festival", list(seasons.keys()))
selected_month = st.sidebar.selectbox("Select month", seasons[selected_season])

# Filter & Model
product_data = data[data['Product Name'] == selected_product][['Month', 'Monthly_Sales']].rename(columns={'Month': 'ds', 'Monthly_Sales': 'y'}).dropna()
model = Prophet(growth='linear')
model.fit(product_data)
future = model.make_future_dataframe(periods=12, freq='MS')
forecast = model.predict(future)

# Clip negatives
forecast['yhat'] = forecast['yhat'].clip(lower=0).round().astype(int)
forecast['yhat_lower'] = forecast['yhat_lower'].clip(lower=0).round().astype(int)
forecast['yhat_upper'] = forecast['yhat_upper'].clip(lower=0).round().astype(int)

# Future data
forecast_data = forecast[forecast['ds'] > product_data['ds'].max()]
forecast_data['Month'] = forecast_data['ds'].dt.strftime('%B')
season_forecast_data = forecast_data[forecast_data['Month'].isin(seasons[selected_season])] if selected_season != "None" else forecast_data
selected_month_forecast_data = season_forecast_data[season_forecast_data['Month'] == selected_month]

# Title & Metrics
st.title("📈 Product Demand Forecasting")
st.write(f"Forecast for **{selected_product}** - **{selected_season} ({selected_month})**")

if not selected_month_forecast_data.empty:
    pred = selected_month_forecast_data['yhat'].iloc[0]
    lower = selected_month_forecast_data['yhat_lower'].iloc[0]
    upper = selected_month_forecast_data['yhat_upper'].iloc[0]
    
    col1, col2, col3 = st.columns(3)
    col1.metric("Prediction", pred)
    col2.metric("Lower", lower)
    col3.metric("Upper", upper)

# Charts
fig = px.line(pd.concat([product_data.assign(yhat='Historical'), forecast.assign(yhat='Forecast')]), 
              x='ds', y='yhat', color='yhat', title=f"{selected_product} Sales Forecast")
st.plotly_chart(fig, use_container_width=True)

st.dataframe(selected_month_forecast_data[['ds', 'yhat', 'yhat_lower', 'yhat_upper']].round(0).astype(int))
